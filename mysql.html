 启动   service mysql start
 
   ==============================
sudo   mysql -uroot -p
 创建用户
 格式（方括号子句可选）

 GRANT privileges [columns]              privileges 权限   列名
 on item                                 数据库或表  .
 to username [IDENTIFIED BY 'password']   
 [REQYURE ssl_options]                   连接选项 
 [WITH [GRANT OPTION | limit_iptions]]   其他选项
 
 例：
 GRANT  all
 on *
 to  k identified by 'password';

 开启远程连接
 use mysql;
 select user,host from user;
 显示：
 +------+-----------+
 | user | host      |
 +------+-----------+
 | k    | %         |
 | root | %         |
 | root | 127.0.0.1 |
 | root | ::1       |
 | root | ki        |
 +------+-----------+
 k 为新添加用户   
 更改用户的host 为 ‘%’
 重启

show tables;              显示表 

创建表
create table server_table
(   ID int unsigned not null auto_increment primary key, 
    service_name char(50) not null, 
    IP char(20)    not null, 
    port char(10)  not null,
    date date      not null
);
+--------------+------------------+------+-----+---------+----------------+
| Field        | Type             | Null | Key | Default | Extra          |
+--------------+------------------+------+-----+---------+----------------+
| ID           | int(10) unsigned | NO   | PRI | NULL    | auto_increment |
| service_name | char(50)         | NO   |     | NULL    |                |
| IP           | char(20)         | NO   |     | NULL    |                |
| port         | char(10)         | NO   |     | NULL    |                |
+--------------+------------------+------+-----+---------+----------------+

desc server_table   显示表结构
